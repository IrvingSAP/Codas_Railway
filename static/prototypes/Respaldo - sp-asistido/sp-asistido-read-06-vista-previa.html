6 · Script generado (READ · result set + OUT · §9.7)

Ejemplo: DYNAMIC RESULT SETS 1, cursor de retorno, mismos P_ERROR_CODE / P_ERROR_MSG que en ADD, sin datos tabulares en el mensaje de error. Síntaxis DB2 for i orientativa. Ver paso 5 y docs/CODAS_SP_ASISTIDO.md §9.7.

-- CODAS · SP asistido · READ · demo
-- Tabla: R1DATA.CUSMAS — datos devueltos vía result set, no vía P_ERROR_*
-- Plantilla: P_ERROR_*, handlers §9, cursor WITH RETURN (§9.7)

CREATE OR REPLACE PROCEDURE R1DATA.SELCUS00 (
    IN  P_ESTADO   CHAR(1)     CCSID 284,
    OUT P_ERROR_CODE   INTEGER,
    OUT P_ERROR_MSG    VARCHAR(200)
)
LANGUAGE SQL
READS SQL DATA
DYNAMIC RESULT SETS 1
SPECIFIC SELCUS0SP1
BEGIN
    /* Ejemplo: un predicado = paso 3 (CUSTATUS + Parámetro IN P_ESTADO) */
    DECLARE c_result CURSOR WITH HOLD
        WITH RETURN FOR
        SELECT CUSNBR, CUSNAME, CUSTATUS, CUSZIPC, CUSCTRY
        FROM   R1DATA.CUSMAS
        WHERE  CUSTATUS = P_ESTADO
        ORDER BY CUSNBR ASC
        FETCH FIRST 100 ROWS ONLY;

    DECLARE v_sqlstate CHAR(5)     DEFAULT '00000';
    DECLARE v_msg        VARCHAR(200) DEFAULT '';

    /* Duplicado (menos frecuente en SELECT; incluido por homogeneidad §9) */
    DECLARE EXIT HANDLER FOR SQLSTATE '23505'
    BEGIN
        SET P_ERROR_CODE = 100;
        SET P_ERROR_MSG  = 'Restricción de unicidad.';
    END;

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        GET DIAGNOSTICS CONDITION 1
            v_sqlstate = RETURNED_SQLSTATE,
            v_msg = MESSAGE_TEXT;
        SET P_ERROR_CODE = -1;
        SET P_ERROR_MSG  = 'Error SQLSTATE=' CONCAT v_sqlstate CONCAT ' - ' CONCAT v_msg;
    END;

    OPEN c_result;
    SET P_ERROR_CODE = 0;
    SET P_ERROR_MSG  = 'OK';
    /* Llamador consume el result set; columnas = paso 2, WHERE/ORDER = pasos 3–4 */
END
