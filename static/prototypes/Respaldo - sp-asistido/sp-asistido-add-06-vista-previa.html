Ejemplo alineado a docs/CODAS_SP_ASISTIDO.md §9 (vista previa) y a los campos de identificación de los paso 3 / 5. En producto, salida de services/sp_sql o equivalente, con P_ERROR_*, SPECIFIC y comentarios ajustables.

-- CODAS · SP asistido · ADD · demo
-- Tabla: R1DATA.CUSMAS (diseño CUSMAS) · CUSNBR: IDENTITY (excl. INSERT)
-- Plantilla: OUT estándar + manejadores 23505 y SQLEXCEPTION (CODAS_SP_ASISTIDO §9)

CREATE OR REPLACE PROCEDURE R1DATA.INSCUS00 (
    IN  P_CUSNAME      VARCHAR(40) CCSID 284,
    IN  P_CUSTATUS     CHAR(1)     CCSID 284,
    OUT P_ERROR_CODE   INTEGER,
    OUT P_ERROR_MSG    VARCHAR(200)
)
LANGUAGE SQL
MODIFIES SQL DATA
SPECIFIC INSCUS0SP1
BEGIN
    /* ========= variables internas (diagnóstico) — §9.4 ========= */
    DECLARE v_sqlstate CHAR(5)     DEFAULT '00000';
    DECLARE v_msg        VARCHAR(200) DEFAULT '';

    /* ========= manejador: clave duplicada (§9.5 / §9.6) ========= */
    DECLARE EXIT HANDLER FOR SQLSTATE '23505'
    BEGIN
        SET P_ERROR_CODE = 100;
        SET P_ERROR_MSG  = 'El registro ya existe.';
    END;

    /* ========= manejador: error SQL general (§9.5) ========= */
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        GET DIAGNOSTICS CONDITION 1
            v_sqlstate = RETURNED_SQLSTATE,
            v_msg = MESSAGE_TEXT;
        SET P_ERROR_CODE = -1;
        SET P_ERROR_MSG  = 'Error SQLSTATE=' CONCAT v_sqlstate CONCAT ' - ' CONCAT v_msg;
    END;

    /* ========= validaciones lógicas (opcionales) — podría quedar vacío (§9.4) ========= */

    /* ========= INSERT (columnas = selección asistente; NULL según C.2/C.3) ========= */
    INSERT INTO R1DATA.CUSMAS (
        CUSNAME, CUSTATUS, CUSZIPC
    )
    VALUES (
        P_CUSNAME, P_CUSTATUS, NULL
    );

    /* ========= finalización OK (camino normal) — §9.2 ========= */
    SET P_ERROR_CODE = 0;
    SET P_ERROR_MSG  = 'OK';
END
